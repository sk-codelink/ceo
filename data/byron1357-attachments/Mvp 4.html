function quotaMiddleware({ quotaKey, amountFn }) {
  return (req, res, next) => {
    const apiKey   = req.header("x-api-key") || "anon";
    const userTier = req.header("x-user-tier") || "free";
    if (userTier !== "free") return next();

    const u     = getUsage(apiKey);
    const used  = u[quotaKey + "Used"];
    const limit = config.FREE_QUOTAS[quotaKey];
    const cost  = amountFn(req);

    if (used + cost > limit) {
      return res.status(402).json({
        error: `Quota exceeded for free tier`,
        quota: limit,
        used
      });
    }
    u[quotaKey + "Used"] = used + cost;
    next();
  };
}

import { useState } from 'react';

export default function ChatBuilder() {
  const questions = [
    "What's your name?",
    "What's your favorite color?",
    "Name a few hobbies or favorite foods."
  ];
  const [step, setStep] = useState(0);
  const [prefs, setPrefs] = useState({});
  const [chatLog, setChatLog] = useState([{ from:'bot', text:questions[0] }]);

  const handleUserAnswer = async (answer) => {
    setChatLog(log=>[...log, { from:'user', text:answer }]);
    setPrefs(p=>({ ...p, [questions[step]]: answer }));
    if (step+1 < questions.length) {
      setStep(s=>s+1);
      setChatLog(log=>[...log, { from:'bot', text:questions[step+1] }]);
    } else {
      const html = await askEvo('generate_homepage', prefs);
      setChatLog(log=>[...log, { from:'bot', text:'Hereâ€™s your page!' }]);
      document.getElementById('preview').innerHTML = html;
    }
  };

  return (
    <div>
      <div id="chat-window">
        {chatLog.map((m,i)=><div key={i} className={m.from}>{m.text}</div>)}
        {step < questions.length && (
          <UserInput onSubmit={handleUserAnswer} />
        )}
      </div>
      <div id="preview" style={{ marginTop:'2rem' }} />
    </div>
  );
}

import Anthropic from '@anthropic-ai/sdk';
const anthropic = new Anthropic({ apiKey: process.env.ANTHROPIC_API_KEY });

export async function askEvo(task, prefs) {
  if (task === 'generate_homepage') {
    const prompt = `
      Create a complete, responsive HTML+CSS homepage.
      - Name: ${prefs["What's your name?"]}
      - Color: ${prefs["What's your favorite color?"]}
      - Interests: ${prefs["Name a few hobbies or favorite foods."]}
    `;
    const response = await anthropic.chat.completions.create({
      model: 'claude-4-opus',
      messages: [
        { role: 'system', content: 'You generate clean, self-contained HTML/CSS.' },
        { role: 'user',   content: prompt }
      ],
      max_tokens_to_sample: 800,
      temperature: 0.7
    });
    return response.completion;
  }
  return '';
}
